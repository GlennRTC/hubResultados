Then create supabase/migrations/0001_patients.sql with RLS policies for the patients table:

```sql
-- Enable RLS
ALTER TABLE patients ENABLE ROW LEVEL SECURITY;
ALTER TABLE orders ENABLE ROW LEVEL SECURITY;
ALTER TABLE order_items ENABLE ROW LEVEL SECURITY;

-- patients: scoped to authenticated lab staff via lab_users.laboratory_id
CREATE POLICY "lab_staff_select_patients" ON patients
  FOR SELECT USING (
    laboratory_id IN (
      SELECT laboratory_id FROM lab_users WHERE auth_user_id = auth.uid()
    )
  );

CREATE POLICY "lab_staff_insert_patients" ON patients
  FOR INSERT WITH CHECK (
    laboratory_id IN (
      SELECT laboratory_id FROM lab_users WHERE auth_user_id = auth.uid()
    )
  );

CREATE POLICY "lab_staff_update_patients" ON patients
  FOR UPDATE USING (
    laboratory_id IN (
      SELECT laboratory_id FROM lab_users WHERE auth_user_id = auth.uid()
    )
  );

-- orders: same pattern
CREATE POLICY "lab_staff_select_orders" ON orders
  FOR SELECT USING (
    laboratory_id IN (
      SELECT laboratory_id FROM lab_users WHERE auth_user_id = auth.uid()
    )
  );

CREATE POLICY "lab_staff_insert_orders" ON orders
  FOR INSERT WITH CHECK (
    laboratory_id IN (
      SELECT laboratory_id FROM lab_users WHERE auth_user_id = auth.uid()
    )
  );

CREATE POLICY "lab_staff_update_orders" ON orders
  FOR UPDATE USING (
    laboratory_id IN (
      SELECT laboratory_id FROM lab_users WHERE auth_user_id = auth.uid()
    )
  );

-- order_items: scoped via join to orders.laboratory_id
CREATE POLICY "lab_staff_select_order_items" ON order_items
```

Create src/app/dashboard/pacientes/nuevo/actions.ts:
```
"use server"
import { redirect } from "next/navigation";
import { db } from "@/lib/db";
import { patients } from "@/lib/db/schema";
import { getLabUser } from "@/lib/auth/get-lab-user";

export async function createPatientAction(formData: FormData) {
  const { lab } = await getLabUser();

  const documentType = formData.get("documentType") as string;
  const documentNumber = (formData.get("documentNumber") as string)?.trim();
  const firstName = (formData.get("firstName") as string)?.trim();
  const lastName = (formData.get("lastName") as string)?.trim();
  const dateOfBirth = formData.get("dateOfBirth") as string; // "YYYY-MM-DD"
  const phone = (formData.get("phone") as string)?.trim() || null;

  if (!documentType || !documentNumber || !firstName || !lastName || !dateOfBirth) {
    redirect("/dashboard/pacientes/nuevo?error=campos_requeridos");
  }

  const validDocTypes = ["CC", "CE", "PA", "RC", "TI"];
  if (!validDocTypes.includes(documentType)) {
    redirect("/dashboard/pacientes/nuevo?error=tipo_documento_invalido");
  }

  await db.insert(patients).values({
    laboratoryId: lab.id,
    documentType: documentType as "CC" | "CE" | "PA" | "RC" | "TI",
    documentNumber,
    firstName,
    lastName,
    dateOfBirth,
    phone,
  });

  redirect("/dashboard/pacientes");
}
```

- Queries patients with Drizzle:
  ```typescript
  let query = db.select().from(patients).where(eq(patients.laboratoryId, lab.id));
  if (q) {
    query = db.select().from(patients).where(
      and(
        eq(patients.laboratoryId, lab.id),
        or(
          ilike(patients.firstName, `%${q}%`),
          ilike(patients.lastName, `%${q}%`),
          ilike(patients.documentNumber, `%${q}%`)
        )
  ```
