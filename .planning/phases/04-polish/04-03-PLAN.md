Create `src/lib/plan-limits.ts`:
```typescript
import { db } from "@/lib/db";
import { laboratories } from "@/lib/db/schema";
import { eq, and, or, lt, ne, sql } from "drizzle-orm";

export interface LimitCheckResult {
  allowed: boolean;
  reason?: "plan_limit";
}

/**
 * Atomically checks the free plan limit and increments the counter if allowed.
 * Uses a single conditional UPDATE...RETURNING to avoid TOCTOU race conditions.
 * Call this BEFORE generating PDF in validateAndSendAction.
 */
export async function checkAndIncrementMonthlyLimit({
  labId,
  plan,
  resultsThisMonth,
}: {
  labId: string;
  plan: "free" | "pro";
  resultsThisMonth: number;
}): Promise<LimitCheckResult> {
  // Pro plan: always allowed — no DB write needed
  if (plan !== "free") {
    return { allowed: true };
  }

  // Free plan: atomic conditional increment
  // Only increments if results_this_month < 30 (prevents going over)
  const updated = await db
    .update(laboratories)
    .set({ resultsThisMonth: sql`results_this_month + 1` })
    .where(
      and(
        eq(laboratories.id, labId),
        lt(laboratories.resultsThisMonth, 30)
      )
    )
    .returning({ resultsThisMonth: laboratories.resultsThisMonth });

  if (updated.length === 0) {
    // No row updated → limit already at 30
    return { allowed: false, reason: "plan_limit" };
  }

  return { allowed: true };
}
```

```typescript
import { redirect } from "next/navigation";

/**
 * Guard for superadmin routes.
 * Checks that the authenticated user's email is in SUPERADMIN_EMAILS env var.
 * Independent of getLabUser() — do NOT combine these guards.
 */
export async function getSuperadmin() {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();

  if (!user) {
    redirect("/login");
  }

  const allowed = (process.env.SUPERADMIN_EMAILS ?? "")
    .split(",")
    .map((e) => e.trim().toLowerCase())
    .filter(Boolean);

  if (!allowed.includes((user.email ?? "").toLowerCase())) {
    redirect("/dashboard");
  }

  return user;
}
```

```typescript
"use server";
import { getSuperadmin } from "@/lib/auth/get-superadmin";
import { createAdminClient } from "@/lib/supabase/server";
import { db } from "@/lib/db";
import { laboratories, labUsers } from "@/lib/db/schema";
import { eq } from "drizzle-orm";
import { redirect } from "next/navigation";

export async function deleteLabAction(formData: FormData) {
  await getSuperadmin();
  const labId = formData.get("labId") as string;
  if (!labId) return;

  const adminClient = createAdminClient();

  // 1. Delete Storage files (GDPR/Ley 1581 erasure)
  // Delete all PDFs in results/{labId}/ prefix
  const { data: resultFiles } = await adminClient.storage.from("results").list(labId);
  if (resultFiles?.length) {
    const paths = resultFiles.map((f) => `${labId}/${f.name}`);
    await adminClient.storage.from("results").remove(paths);
  }
  // Delete logo
  const exts = ["png", "jpg", "jpeg", "webp"];
  for (const ext of exts) {
    await adminClient.storage.from("logos").remove([`logos/${labId}.${ext}`]);
  }

  // 2. Delete all lab_users auth accounts
  const staffRows = await db.select({ authUserId: labUsers.authUserId }).from(labUsers).where(eq(labUsers.laboratoryId, labId));
  for (const { authUserId } of staffRows) {
    try {
      await adminClient.auth.admin.deleteUser(authUserId);
    } catch { /* Continue even if one fails */ }
  }

  // 3. Delete lab row — cascades to all child tables via FK
  await db.delete(laboratories).where(eq(laboratories.id, labId));

  redirect("/superadmin");
}
```
