Create `src/app/dashboard/configuracion/actions.ts`:
- `"use server"` at top
- Import: getLabUser, db, laboratories, labUsers, eq, and, revalidatePath, z, createAdminClient, uploadLogo
- **updateLabSettingsAction(formData: FormData)**:
  - Call getLabUser(); if user.role !== "admin" return { error: "No autorizado" }
  - Zod schema: name z.string().min(2).max(100), whatsappPhoneId z.string().optional(), whatsappToken z.string().optional()
  - If safeParse fails, return { error: "Datos inválidos" }
  - db.update(laboratories).set(parsed.data).where(eq(laboratories.id, lab.id))
  - revalidatePath("/dashboard/configuracion")
  - return { success: true }
- **uploadLogoAction(formData: FormData)**:
  - getLabUser(); if user.role !== "admin" return { error: "No autorizado" }
  - Get file from formData.get("logo") as File; if not File or size > 2MB return { error }
  - Only accept image/png, image/jpeg, image/webp content types
  - Convert File to Buffer via arrayBuffer()
  - Call uploadLogo(buffer, lab.id, file.type)
  - db.update(laboratories).set({ logoUrl: publicUrl }).where(eq(laboratories.id, lab.id))
  - revalidatePath("/dashboard/configuracion")
  - return { success: true, logoUrl: publicUrl }
- **inviteStaffAction(formData: FormData)**:
  - getLabUser(); if user.role !== "admin" return { error: "No autorizado" }
  - email = (formData.get("email") as string)?.trim().toLowerCase()
  - role = formData.get("role") — must be one of "admin" | "technician" | "reception"
  - If invalid email or invalid role return { error: "Datos inválidos" }
  - Check for duplicate: db.select().from(labUsers).where(and(eq(labUsers.email, email), eq(labUsers.laboratoryId, lab.id))) — if found return { error: "El usuario ya pertenece a este laboratorio" }
  - createAdminClient().auth.admin.inviteUserByEmail(email, { data: { laboratory_id: lab.id, role }, redirectTo: `${process.env.NEXT_PUBLIC_APP_URL}/auth/callback?next=/dashboard` })
  - If Supabase returns error: return { error: "No se pudo enviar la invitación" }
  - db.insert(labUsers).values({ laboratoryId: lab.id, email, fullName: email, role, authUserId: data.user.id })
  - revalidatePath("/dashboard/configuracion")
  - return { success: true }
- **removeStaffAction(formData: FormData)**:
  - getLabUser(); if user.role !== "admin" return { error: "No autorizado" }
  - staffId = formData.get("staffId") as string
  - Fetch staff row: db.select().from(labUsers).where(and(eq(labUsers.id, staffId), eq(labUsers.laboratoryId, lab.id))).limit(1)
  - If not found: return { error: "Usuario no encontrado" }
  - If staffRow.id === user.id: return { error: "No puedes eliminarte a ti mismo" }
  - createAdminClient().auth.admin.deleteUser(staffRow.authUserId) — if error, return { error }
  - db.delete(labUsers).where(and(eq(labUsers.id, staffId), eq(labUsers.laboratoryId, lab.id)))
  - revalidatePath("/dashboard/configuracion")
  - return { success: true }
