```typescript
import { db } from "@/lib/db";
import { orders, notifications, auditLog } from "@/lib/db/schema";
import { eq, and, gte, count, sql } from "drizzle-orm";

export interface AnalyticsResult {
  resultsToday: number;
  deliveryRate: number;       // 0-1 decimal
  avgTimeMinutes: number;     // avg minutes from validatedAt to first result_viewed audit entry
}

export async function getAnalytics(labId: string): Promise<AnalyticsResult> {
  const now = new Date();
  const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  const monthStart = new Date(now.getFullYear(), now.getMonth(), 1);

  // Results validated today
  const [todayResult] = await db
    .select({ total: count() })
    .from(orders)
    .where(
      and(
        eq(orders.laboratoryId, labId),
        gte(orders.validatedAt, todayStart),
      )
    );

  // Delivery rate this month: delivered / validated
  const [monthStats] = await db
    .select({
      validated: count(),
      delivered: sql<number>`COUNT(*) FILTER (WHERE ${orders.status} = 'delivered' OR ${orders.status} = 'read')`,
    })
    .from(orders)
    .where(
      and(
        eq(orders.laboratoryId, labId),
        gte(orders.validatedAt, monthStart),
      )
    );

  const deliveryRate =
    monthStats.validated > 0
      ? Number(monthStats.delivered) / monthStats.validated
      : 0;

  // Avg time from validated_at to first result_viewed event in audit_log
  const [avgResult] = await db
    .select({
      avgMinutes: sql<number>`
        COALESCE(
          AVG(
            EXTRACT(EPOCH FROM (${auditLog.createdAt} - ${orders.validatedAt})) / 60
          ),
          0
        )
      `.as("avgMinutes"),
    })
    .from(auditLog)
    .innerJoin(orders, eq(auditLog.targetId, orders.id))
    .where(
      and(
        eq(auditLog.laboratoryId, labId),
        eq(auditLog.action, "result_viewed"),
      )
    );

  return {
    resultsToday: todayResult?.total ?? 0,
    deliveryRate: Math.min(1, Math.max(0, deliveryRate)),
    avgTimeMinutes: Number(avgResult?.avgMinutes ?? 0),
  };
}
```

```typescript
import { db } from "@/lib/db";
import { auditLog, labUsers } from "@/lib/db/schema";
import { eq, desc, count } from "drizzle-orm";

const PAGE_SIZE = 25;

export interface AuditEntry {
  id: string;
  action: string;
  targetId: string | null;
  ipAddress: string | null;
  createdAt: Date;
  laboratoryId: string;
  userName: string | null;
}

export interface AuditLogResult {
  entries: AuditEntry[];
  totalPages: number;
  page: number;
}

export async function getAuditLog({
  labId,
  page,
}: {
  labId: string;
  page: number;
}): Promise<AuditLogResult> {
  const safePageNumber = Math.max(1, page);
  const offset = (safePageNumber - 1) * PAGE_SIZE;

  const [{ total }] = await db
    .select({ total: count() })
    .from(auditLog)
    .where(eq(auditLog.laboratoryId, labId));

  const entries = await db
    .select({
      id: auditLog.id,
      action: auditLog.action,
      targetId: auditLog.targetId,
      ipAddress: auditLog.ipAddress,
      createdAt: auditLog.createdAt,
      laboratoryId: auditLog.laboratoryId,
      userName: labUsers.fullName,
    })
    .from(auditLog)
    .leftJoin(labUsers, eq(auditLog.userId, labUsers.id))
    .where(eq(auditLog.laboratoryId, labId))
    .orderBy(desc(auditLog.createdAt))
    .limit(PAGE_SIZE)
    .offset(offset);

  return {
    entries,
    totalPages: Math.max(1, Math.ceil(total / PAGE_SIZE)),
    page: safePageNumber,
  };
}
```

```typescript
export async function resendNotificationAction(formData: FormData) {
  const { lab, user } = await getLabUser();
  const orderId = (formData.get("orderId") as string)?.trim();

  if (!orderId) return { error: "ID de orden requerido" };

  // Fetch order — must belong to this lab and be validated (not pending)
  const [order] = await db
    .select()
    .from(orders)
    .where(and(eq(orders.id, orderId), eq(orders.laboratoryId, lab.id)))
    .limit(1);

  if (!order) return { error: "Orden no encontrada" };
  if (order.status === "pending") return { error: "La orden no ha sido validada aún" };

  // Fetch patient phone
  const [patient] = await db
    .select()
    .from(patients)
    .where(and(eq(patients.id, order.patientId), eq(patients.laboratoryId, lab.id)))
    .limit(1);

  if (!patient?.phone) return { error: "El paciente no tiene teléfono registrado" };

  // Attempt WhatsApp send — use per-lab credentials if configured, else env vars
  let wamid: string | null = null;
  try {
    wamid = await sendResultadoListo(
      patient.phone,
      patient.firstName,
      lab.name,
      order.verificationCode
    );
  } catch {
    // Send failed — will insert failed notification below
  }

  // Always insert a new notifications row (preserves full history)
  await db.insert(notifications).values({
    orderId,
    laboratoryId: lab.id,
    channel: "whatsapp",
    status: wamid ? "sent" : "failed",
    whatsappMessageId: wamid,
  });

  // Advance order status on successful send
  if (wamid) {
    await db
      .update(orders)
      .set({ status: "delivered" })
      .where(eq(orders.id, orderId));
  }

  // Audit trail
  await db.insert(auditLog).values({
    laboratoryId: lab.id,
    userId: user.id,
    action: "notification_resent",
    targetId: orderId,
  });

  revalidatePath(`/dashboard/ordenes/${orderId}`);
  return { success: true, sent: !!wamid };
}
```
